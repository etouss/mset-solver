(set-logic QF_SLRDI)
(declare-fun emptybag () BagInt)(declare-fun M0 () BagInt)
(declare-fun M1 () BagInt)
(declare-fun M2 () BagInt)
(declare-fun key () Int)
(declare-fun d1 () Int)
(assert (and 
 	(= M2 emptybag )
 	(= M0 (bagunion (bag d1 ) M1 M2 ) )
 	(< M1 (bag d1 ) )
 	(< (bag d1 ) M2 )
 	(= key d1 )
 )
 ) 
 (assert (not 
(and 
 	(= M1 (bagminus M0 (bag key ) ) )
 	(subset (bag key ) M0 )
 )
 ) 
)
(check-sat)
